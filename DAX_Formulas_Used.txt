
üìò DAX Measures Used in Power BI Project

üìÖ Year-wise Profit Calculation
---------------------------------
2016 profit = CALCULATE([Total_Profit], Store[Year] = 2016)
2017 profit = CALCULATE([Total_Profit], Store[Year] = 2017)

üïì Time-Based Measures
------------------------
Average Delivery Time = AVERAGE(Store[Datediff])
DATESBETWEEN = CALCULATE([Total_Profit], DATESBETWEEN(calender[Date], "24-09-2017", "30-09-2017"))
DATESINPERIOD = CALCULATE([Total_Profit], DATESINPERIOD(calender[Date], "24-09-2014", 7, DAY))
YTD = TOTALYTD([Total_Profit], calender[Date])
QTD = TOTALQTD([Total_Profit], calender[Date])
sameperiodlastyear = CALCULATE([Total_Profit], SAMEPERIODLASTYEAR(calender[Date]))
Parallel Period profit = CALCULATE([Total_Profit], PARALLELPERIOD(calender[Date], -1, MONTH))
Parallel Period sales = CALCULATE([Total_Sales], PARALLELPERIOD(calender[Date], -1, MONTH))

üìä Category-Based Sales & Profit
----------------------------------
Sales of Furniture = CALCULATE([Total_Sales], Store[Category] = "Furniture")
Sales of Supplies = CALCULATE([Total_Sales], Store[Category] = "Office Supplies")
Sales of Technology = CALCULATE([Total_Sales], Store[Category] = "Technology")
supplies_profit = CALCULATE([Total_Profit], Store[Category] = "Office Supplies")

üìà Contribution Calculations
------------------------------
CALCULATE = CALCULATE([Total_Sales], ALL(Store[Category]))
Contribution by profit = [Total_Profit] / [CALCULATE]
Contribution by sales = [Total_Sales] / [CALCULATE]
Remove_Filters = CALCULATE([Total_Sales], REMOVEFILTERS(Store[Sub-Category]))
Total Profit Contribution of each sub_Category = [Total_Profit] / [Remove_Filters]
Total Sales Contribution of each sub_Category = [Total_Sales] / [Remove_Filters]

üìå Total KPIs
---------------
Total_Profit = SUM(Store[Profit])
Total_Sales = SUM(Store[Sales])
Sales/profit% = [Total_Profit] / [Total_Sales]

üì¶ Count Measures
-------------------
countrows = COUNTROWS(Store)
Total_Quantity_Sold_across_all_Office_Supplies = COUNTROWS(FILTER('Store', Store[Category] = "Office Supplies"))
Total_Quantity_Sold_across_all_Technology = COUNTROWS(FILTER('Store', Store[Category] = "Technology"))
Total_Quantity_Sold_across_Furniture = COUNTROWS(FILTER('Store', Store[Category] = "Furniture"))

üèÜ Ranking
-----------
Rank = RANKX(ALL(Store[State]), [Total_Profit], , DESC)
